;Пример пользовательского словаря
; Знаком + перед гласной обозначается ударение.

[dict]
; Произношение отдельного слова.
себе = себ+е
тебе = теб+е
еще = ещё

; Произношение слова с учётом регистра букв.
Тому/c = т+ому
; Произношение этого слова по умолчанию.
тому = том+у

; Произношение части слова.
; Нужно обязательно указать как префикс, так и суффикс, которые должны
; представлять из себя регулярные выражения, составленные согласно
; синтаксису, поддерживаемому библиотекой pcre.
.* интернет .* = интэрн+эт

; Произношение символов.
#169; = знак авторского права
#x2116; = номер
